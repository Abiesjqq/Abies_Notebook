    print_expr <- function(expr) {
  e <- substitute(expr) # 捕获原始表达式
  v <- eval(e, envir = parent.frame()) # 计算

  # 确保表达式显示为一行
  expr_str <- paste(deparse(e), collapse = " ")

  cat(expr_str, "= ")

  if (is.atomic(v) && length(v) > 1) {
    cat(v, sep = " ")
    cat("\n")
  } else if (is.matrix(v) || is.data.frame(v)) {
    print(v)
  } else {
    print(v)
  }

  invisible(v)
}
